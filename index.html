async function startAdaptiveEngine() { if(isRunning) return; // RESET LOG UI out("INIZIALIZZAZIONE...", "pulse"); try { const rpcUrl = "https://testnet-passet-hub-eth-rpc.polkadot.io"; const pKey = document.getElementById('pKey').value.trim(); const cAddr = document.getElementById('contractAddr').value.trim(); if(!pKey.startsWith("0x")) throw new Error("La Private Key deve iniziare con 0x"); if(!ethers.isAddress(cAddr)) throw new Error("Indirizzo Contratto non valido"); const provider = new ethers.JsonRpcProvider(rpcUrl); // Verifica connessione RPC const network = await provider.getNetwork(); out(`Connesso a Chain ID: ${network.chainId}`, "void"); const wallet = new ethers.Wallet(pKey, provider); out(`Wallet pronto: ${wallet.address.substring(0,10)}...`); isRunning = true; document.getElementById('statusInfo').innerText = "STATUS: ONLINE - ATTACKING"; // Lancio del loop loop(provider, wallet, cAddr); } catch (e) { out(`ERRORE CRITICO ALL'AVVIO: ${e.message}`, "danger"); console.error(e); } }
